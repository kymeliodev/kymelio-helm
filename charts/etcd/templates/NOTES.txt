Run etcdctl from a temporary client pod:

  kubectl run etcdctl --rm -it --restart=Never \
    --namespace {{ .Release.Namespace }} \
    --image=quay.io/coreos/etcd:{{ .Values.image.tag | default .Chart.AppVersion }} -- \
    sh -c 'ETCDCTL_API=3 etcdctl --endpoints http://{{ include "etcd.fullname" . }}:{{ .Values.service.ports.client.port }} endpoint health'

  ETCDCTL_API=3 etcdctl --endpoints http://{{ include "etcd.fullname" . }}:{{ .Values.service.ports.client.port }} put foo bar
  ETCDCTL_API=3 etcdctl --endpoints http://{{ include "etcd.fullname" . }}:{{ .Values.service.ports.client.port }} get foo
