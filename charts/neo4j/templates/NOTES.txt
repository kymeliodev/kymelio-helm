  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "neo4j.fullname" . }} {{ .Values.service.ports.bolt.port }}:{{ .Values.service.ports.bolt.port }} {{ .Values.service.ports.http.port }}:{{ .Values.service.ports.http.port }}

Then open the browser at http://127.0.0.1:{{ .Values.service.ports.http.port }}
